<h2>View Team Fixtures</h2>

<select id="teamFilter" onchange="renderFixtures()">

<option value="all">All Teams</option>

</select>

<th>GF</th><th>GA</th><th>Pts</th>

</tr>

</thead>

<tbody id="table"></tbody>

</table>

<h2>Fixtures & Results</h2>

<table>

<thead>

<tr><th>Home</th><th>Away</th><th>Score</th><th>Action</th></tr>

</thead>

<script>

let teams=[],fixtures=[];


function addTeam(){

let n=teamName.value.trim();

if(!n||teams.find(t=>t.name===n))return;

teams.push({name:n,P:0,W:0,D:0,L:0,GF:0,GA:0,PTS:0});

teamName.value="";

updateSelect();generateFixtures();renderTable();renderFixtures();

}


function updateSelect(){

teamFilter.innerHTML="<option value='all'>All Teams</option>";

teams.forEach(t=>{

let o=document.createElement("option");

o.value=t.name;o.textContent=t.name;

teamFilter.appendChild(o);

});

}


function generateFixtures(){

fixtures=[];

for(let i=0;i<teams.length;i++)

for(let j=i+1;j<teams.length;j++){

fixtures.push({h:teams[i].name,a:teams[j].name,hs:null,as:null});

fixtures.push({h:teams[j].name,a:teams[i].name,hs:null,as:null});

}

}


function renderTable(){

table.innerHTML="";

teams.sort((a,b)=>b.PTS-a.PTS||(b.GF-b.GA)-(a.GF-a.GA));

teams.forEach(t=>{

table.innerHTML+=`<tr>

<td>${t.name}</td><td>${t.P}</td><td>${t.W}</td>

<td>${t.D}</td><td>${t.L}</td>

<td>${t.GF}</td><td>${t.GA}</td><td>${t.PTS}</td>

</tr>`;

});

}


function renderFixtures(){

fixturesEl=fixturesEl||document.getElementById("fixtures");

fixturesEl.innerHTML="";

fixtures.filter(f=>teamFilter.value==="all"||f.h===teamFilter.value||f.a===teamFilter.value)

.forEach((f,i)=>{

fixturesEl.innerHTML+=`

<tr>

<td>${f.h}</td><td>${f.a}</td>

<td><input size=2 id=h${i} value="${f.hs??""}"> -

<input size=2 id=a${i} value="${f.as??""}"></td>

<td><button onclick="save(${i})">Save</button></td>

</tr>`;

});

}


function reset(){

teams.forEach(t=>Object.assign(t,{P:0,W:0,D:0,L:0,GF:0,GA:0,PTS:0}));

}


function save(i){

let hs=parseInt(document.getElementById("h"+i).value),

as=parseInt(document.getElementById("a"+i).value);

if(isNaN(hs)||isNaN(as))return;

fixtures[i].hs=hs;fixtures[i].as=as;

reset();

fixtures.forEach(f=>{

if(f.hs==null)return;

let h=teams.find(t=>t.name===f.h),

a=teams.find(t=>t.name===f.a);

h.P++;a.P++;h.GF+=f.hs;h.GA+=f.as;a.GF+=f.as;a.GA+=f.hs;

if(f.hs>f.as){h.W++;h.PTS+=3;a.L++}

else if(f.hs<f.as){a.W++;a.PTS+=3;h.L++}

else{h.D++;a.D++;h.PTS++;a.PTS++}

});

renderTable();

}

</script>
